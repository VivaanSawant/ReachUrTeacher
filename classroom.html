import js from pyscript import document, display import asyncio API_BASE_URL = "http://localhost:5000" async def fetch_raised_hands(): while True: response = await js.fetch(f"{API_BASE_URL}/faces_data") text = await response.text() document.getElementById("raisedHands").innerHTML = text await asyncio.sleep(1) async def clear_hands(): await js.fetch(f"{API_BASE_URL}/clear_hands", method="POST") document.getElementById("raisedHands").innerHTML = "No hands raised." asyncio.create_task(fetch_raised_hands())
